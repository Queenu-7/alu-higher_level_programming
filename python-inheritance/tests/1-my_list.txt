#!/usr/bin/python3
MyList = __import__('1-my_list').MyList

def test_print_sorted(capsys):
    my_list = MyList()
    my_list.append(5)
    my_list.append(1)
    my_list.append(3)
    my_list.append(2)
    my_list.append(4)

    # Test that print_sorted prints sorted list
    my_list.print_sorted()
    captured = capsys.readouterr()
    assert captured.out == "[1, 2, 3, 4, 5]\n"

    # Test that the original list is unchanged
    assert my_list == [5, 1, 3, 2, 4]

def test_inheritance():
    my_list = MyList([10, 9, 8])
    assert isinstance(my_list, list)
    assert my_list == [10, 9, 8]

if __name__ == "__main__":
    import sys
    import pytest
    sys.exit(pytest.main([__file__]))
